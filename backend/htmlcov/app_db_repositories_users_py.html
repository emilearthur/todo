100 async def register_new_user(self, *, new_user: UserCreate) -> UserInDB: 

101 """Register new user.""" 

102 if await self.get_user_by_email(email=new_user.email): 

103 raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, 

104 detail=f"{new_user.email} is already taken. Register with a new email") 

105 if await self.get_user_by_username(username=new_user.username): 

106 raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, 

107 detail=f"{new_user.username} is already taken. Register with a new email") 

108 user_pwd_update = self.auth_service.create_salt_and_hashed_password(plaintext_pwd=new_user.password) 

109 new_user_params = new_user.copy(update=user_pwd_update.dict()) 

110 created_user = await self.db.fetch_one(query=REGISTER_NEW_USER_QUERY, values=new_user_params.dict()) 

111 

112 # create profile for new user 

113 await self.profiles_repo.create_profile_for_user(profile_create=ProfileCreate(user_id=created_user["id"])) 

114 return await self.populate_user(user=UserInDB(**created_user))

116 async def authenticate_user(self, *, email: EmailStr, password: str) -> Optional[UserInDB]: 

117 """Authenticate user.""" 

118 user = await self.get_user_by_email(email=email, populate=False) 

119 if not user: 

120 return None 

121 if not self.auth_service.verify_password(pwd=password, salt=user.salt, hashed_pwd=user.password): 

122 return None 

123 return user

137 def generate_otp(self, *, size=6, chars=string.ascii_uppercase + string.digits + string.ascii_lowercase) -> str: 

138 """Generate opt for user verification.""" 

139 return ''.join(random.choice(chars) for _ in range(size))
